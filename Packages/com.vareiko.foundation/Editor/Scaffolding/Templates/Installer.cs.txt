        // MessagePipeOptions to pass as signalOptions.

        {

            builder.RegisterMessageBroker<{{MODULE_NAME}}InitializedSignal>(signalOptions);
            builder.RegisterEntryPoint<{{MODULE_NAME}}Service>(Lifetime.Singleton).As<I{{MODULE_NAME}}Service>().AsSelf();
        }
    }
